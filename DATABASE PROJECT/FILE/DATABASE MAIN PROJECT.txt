             (L_ID NUMBER(10) PRIMARY KEY,
              CITY VARCHAR2(15),
              COUNTRY VARCHAR2(15) );
CREATE TABLE EXHIBITION
             (E_ID NUMBER(10) PRIMARY KEY,
              START_DATE DATE,
              END_DATE DATE,
              G_ID NUMBER(10) );
CREATE TABLE ARTIST
             (A_ID NUMBER(10) PRIMARY KEY,
              A_NAME VARCHAR2(25),
              ADDRESS VARCHAR2(25),
              PHONE VARCHAR2(14) );
CREATE TABLE AG_ID
             (A_ID NUMBER(10),G_ID NUMBER(10),

CREATE TABLE PAINTING
